# ----------------------------------------------------------------------------
# GPIO控制库 (核心依赖)
# ----------------------------------------------------------------------------
# gpiozero: 简单易用的GPIO控制库,支持所有树莓派型号
# 用于控制步进电机(TB6600)、RGB LED、按钮等外设
gpiozero>=2.0.0

# lgpio: gpiozero的底层引脚工厂,替代已废弃的RPi.GPIO
# 在代码中通过 os.environ['GPIOZERO_PIN_FACTORY']='lgpio' 设置
# 注意: 需要先安装系统包 sudo apt-get install python3-lgpio
# 如果pip安装失败,使用: sudo apt-get install python3-lgpio python3-gpiozero
# lgpio>=0.2.2.0

# ----------------------------------------------------------------------------
# I2C通信库 (LCD1602显示)
# ----------------------------------------------------------------------------
# smbus2: 纯Python实现的I2C通信库,用于LCD1602显示屏
smbus2>=0.4.3
colorlog>=6.7.0
# ----------------------------------------------------------------------------
# 可选工具库
# ----------------------------------------------------------------------------
# colorlog: 彩色日志输出,方便调试(可选)
# colorlog>=6.7.0

# 说明
# ----------------------------------------------------------------------------
# 
# 系统依赖 (需要在安装pip包前先安装):
#   sudo apt-get update
#   sudo apt-get install -y python3-pip python3-dev i2c-tools
#   sudo apt-get install -y python3-lgpio python3-gpiozero python3-smbus
#
# 启用I2C接口 (LCD1602必需):
#   sudo raspi-config
#   -> Interface Options -> I2C -> Enable
#   重启后验证: ls /dev/i2c-*
#
# 测试I2C设备 (LCD地址通常是0x27或0x3F):
#   sudo i2cdetect -y 1
#
# GPIO权限 (避免每次都用sudo):
#   sudo usermod -a -G gpio $USER
#   sudo usermod -a -G i2c $USER
#   重新登录后生效
